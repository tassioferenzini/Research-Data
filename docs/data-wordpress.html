# Define factor
var <- Metrics_Wordpress$LOC

# Defines treatment
trat <- Metrics_Wordpress$F_Version

# Descriptive
summary(var)

# Parameter Estimates
xb <- mean(var, na.rm = TRUE) # mu
sx <- sd(var, na.rm = TRUE) # sigma
cat("\n Sample Average =", xb, "\n Standard Deviation =", sx)

# Shapiro
result <- shapiro.test(var)
# Analyze
if (result$p.value > 0.05){
  print(result)
  print ("The data follow a normal distribution.")
} else {
  print(result)
  print ("The data do not follow a normal distribution.")
}

# KS
var <- Metrics_Wordpress$RSYSC
